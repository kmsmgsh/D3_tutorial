import * as d3 from 'd3';

/**
 * Sample data for the bar chart
 * @type {Array<DataPoint>}
 */
const data = [
    { name: 'A', value: 30 },
    { name: 'B', value: 80 },
    { name: 'C', value: 45 },
    { name: 'D', value: 60 },
    { name: 'E', value: 20 },
    { name: 'F', value: 90 }
];

/**
 * Data point for bar chart
 * @typedef {Object} DataPoint
 * @property {string} name - Category name
 * @property {number} value - Numeric value for bar height
 */

/**
 * Chart configuration and dimensions
 * @typedef {Object} ChartConfig
 * @property {Object} margin - Margin configuration
 * @property {number} margin.top - Top margin in pixels
 * @property {number} margin.right - Right margin in pixels
 * @property {number} margin.bottom - Bottom margin in pixels
 * @property {number} margin.left - Left margin in pixels
 * @property {number} width - Chart width in pixels
 * @property {number} height - Chart height in pixels
 */

/** @type {ChartConfig} */
const margin = { top: 20, right: 30, bottom: 40, left: 40 };
const width = 760 - margin.left - margin.right;
const height = 360 - margin.top - margin.bottom;

/**
 * Creates the main SVG container for the chart
 * @returns {d3.Selection} D3 selection of the chart group element
 */
const svg = d3.select("#chart")
    .append("svg")
    .attr("width", width + margin.left + margin.right)
    .attr("height", height + margin.top + margin.bottom)
    .append("g")
    .attr("transform", `translate(${margin.left},${margin.top})`);

/**
 * Creates X scale for positioning bars horizontally
 * @returns {d3.ScaleBand} D3 band scale for categories
 */
const xScale = d3.scaleBand()
    .domain(data.map(d => d.name))
    .range([0, width])
    .padding(0.1);

/**
 * Creates Y scale for positioning bars vertically
 * @returns {d3.ScaleLinear} D3 linear scale for values
 */
const yScale = d3.scaleLinear()
    .domain([0, d3.max(data, d => d.value)])
    .range([height, 0]);

/**
 * Creates the bar chart rectangles
 * Uses D3's data binding pattern to create one rectangle per data point
 */
svg.selectAll(".bar")
    .data(data)
    .enter()
    .append("rect")
    .attr("class", "bar")
    .attr("x", d => xScale(d.name))
    .attr("width", xScale.bandwidth())
    .attr("y", d => yScale(d.value))
    .attr("height", d => height - yScale(d.value));

/**
 * Adds X-axis to the bottom of the chart
 */
svg.append("g")
    .attr("class", "axis")
    .attr("transform", `translate(0,${height})`)
    .call(d3.axisBottom(xScale));

/**
 * Adds Y-axis to the left side of the chart
 */
svg.append("g")
    .attr("class", "axis")
    .call(d3.axisLeft(yScale));

/**
 * Adds Y-axis label (rotated)
 */
svg.append("text")
    .attr("transform", "rotate(-90)")
    .attr("y", 0 - margin.left)
    .attr("x", 0 - (height / 2))
    .attr("dy", "1em")
    .style("text-anchor", "middle")
    .text("Value");

/**
 * Adds X-axis label
 */
svg.append("text")
    .attr("transform", `translate(${width / 2}, ${height + margin.bottom})`)
    .style("text-anchor", "middle")
    .text("Category");
